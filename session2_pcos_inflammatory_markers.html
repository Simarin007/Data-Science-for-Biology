Session 2: PCOS and Inflammatory Markers in Menstrual Cycle

Author

Data Science for Biology Workshop

Research Question

Do inflammatory marker levels differ between participants with PCOS and participants without PCOS after accounting for menstrual-cycle stage?

In this analysis we compare inflammatory cytokine concentrations between participants diagnosed with PCOS and those with no disease, while adjusting for the time point (menstrual-cycle stage) as a covariate.

Mixed-Effects Model

# ============================================================================
# Fit a linear mixed-effects model for each cytokine
# Formula: log_conc ~ pcos_status + time_point + (1 | pid)
#   - pcos_status: fixed effect of interest (PCOS vs no disease)
#   - time_point: adjustment for menstrual-cycle stage
#   - (1 | pid): random intercept per participant (accounts for repeated measures)
# ============================================================================

# Get unique cytokines
cytokines <- unique(analysis_data$cytokine)

# Create a results dataframe
model_results <- data.frame()

for (cyt in cytokines) {
  # Subset data for this cytokine
  sub_data <- analysis_data %>%
    filter(cytokine == cyt) %>%
    mutate(
      # Ensure pcos_status is a factor with "no disease" as reference
      pcos_status = factor(pcos_status, levels = c("no disease", "pcos")),
      # Ensure time_point is a factor for the model
      time_point = factor(time_point, levels = c("week_prior", "onset",
                                                  "end_bleeding", "week_post"))
    )

  # Fit the mixed-effects model
  # Using lmerTest which provides p-values via Satterthwaite approximation
  fit <- lmer(log_conc ~ pcos_status + time_point + (1 | pid),
              data = sub_data)

  # Extract the coefficient for pcos_status (PCOS vs no disease)
  coef_summary <- summary(fit)$coefficients

  # Find the row for pcos_status (it will be "pcos_statuspcos")
  pcos_row <- grep("pcos_status", rownames(coef_summary))

  # Get confidence interval
  ci <- confint(fit, parm = "pcos_statuspcos", method = "Wald")

  # Store results
  row_result <- data.frame(
    cytokine = cyt,
    estimate = coef_summary[pcos_row, "Estimate"],
    std_error = coef_summary[pcos_row, "Std. Error"],
    ci_lower = ci[1],
    ci_upper = ci[2],
    p_value = coef_summary[pcos_row, "Pr(>|t|)"],
    n_obs = nrow(sub_data),
    n_participants = n_distinct(sub_data$pid),
    stringsAsFactors = FALSE
  )

  model_results <- rbind(model_results, row_result)
}

# Apply Benjamini-Hochberg correction across all cytokines
model_results$p_adjusted <- p.adjust(model_results$p_value, method = "BH")

# Sort by adjusted p-value
model_results <- model_results %>%
  arrange(p_adjusted)

# Display results
kable(model_results, digits = 4,
      caption = "Mixed-effects model results: PCOS effect on log cytokine concentration")
